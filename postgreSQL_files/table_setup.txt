-- For registered users
CREATE TABLE reg_users (
	user_id INT PRIMARY KEY,
	last_name VARCHAR,
	first_name VARCHAR,
	state VARCHAR,
	email VARCHAR,
	ntf_method VARCHAR,
	reg_car_1 INT REFERENCES reg_cars,
	reg_car_2 INT REFERENCES reg_cars
);

-- For all car models
CREATE TABLE car_models (
	car_id INT PRIMARY KEY,
	car_manuf VARCHAR,
	car_model VARCHAR,
	car_yr INT
);

-- For safety issues
CREATE TABLE safety_issues (
	issue_id INT PRIMARY KEY,
	description VARCHAR,
	date_known INT,
	ntfs_sent BOOLEAN,
	aff_car1 INT REFERENCES car_models
	aff_car2 INT REFERENCES car_models
	aff_car3 INT REFERENCES car_models

);

-- For registered cars (registered to users)
CREATE TABLE reg_cars (
	reg_car_id INT PRIMARY KEY,
	car_model INT REFERENCES car_models,
	car_owner INT REFERENCES reg_users
);
